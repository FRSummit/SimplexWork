            resizeScreen();
	},

	addBodyClass = function addBodyClass(){

            });   
	},

             }

            jQuery('.blue-cross-icon').click(function() {
              jQuery('.blue-banner-section').css('display', 'none');
            });   
	};
	return {
		init: init
	};
}(window,document,jQuery));
