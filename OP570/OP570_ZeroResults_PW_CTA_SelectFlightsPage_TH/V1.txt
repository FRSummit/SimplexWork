<script type="text/javascript">
  dom.addCss(css);

  window.jetstarOptimisation = window.jetstarOptimisation || {};
  window.jetstarOptimisation.PWCTAColor_SelectFlightsPage_SG= (function (window, document) {

  var $depart = jQuery('.tabs-content.flight-schedule:eq(0)'),
  $return = jQuery('.tabs-content.flight-schedule:eq(1)'),
  $departContainer = jQuery(".new-banner",$depart ),
  $returnContainer = jQuery(".new-banner",$return ),
  
  init = function () {
    addBodyClass();
    FaresGuaranteedHide();
    resizeScreen();
    runLoop();
    variantOne();
    actionTracking();
  },


  addBodyClass = function addBodyClass(){
    jQuery('body').addClass('OP570');
  },

  variantOne = function variantOne() {
    $(document).ready(function() {
      $('input#email').on('keypress', function(e) {
        if(e.which === 13) {
          $(this).parents('.PriceWatch-btn-section').find('#btnValidate').click();
          return false;
        }
      });
    });
  },

  runLoop = function runLoop() {
    jQuery('.content.js-tab-content:visible').each(function(index, element){
      addOffer(element);
    });

    jQuery('.row.new-banner').each(function(index, element){
      bindBlurEvent(element);
      SelectFlightsPage(element);
      showLegal(element);
    })
  },

  showLegal = function showLegal(element) {
     jQuery(element).find('.toggle-legal-show a').on('click', function(){
      jQuery(element).find('.hidden-legal').removeClass('display-hide');
      jQuery(element).find('.toggle-legal-show').addClass('display-hide');
      jQuery(element).find('.toggle-legal-hide').removeClass('display-hide');
    });

    jQuery(element).find('.toggle-legal-hide a').on('click', function(){
      jQuery(element).find('.hidden-legal').addClass('display-hide');
      jQuery(element).find('.toggle-legal-hide').addClass('display-hide');
      jQuery(element).find('.toggle-legal-show').removeClass('display-hide');
    });
  },

  addOffer = function addOffer(element){
    var availableFlights = jQuery(element).find('.flight-card-wrapper:not(.sold-out)').find('.button-toggle').length;
    if (availableFlights == 0) {
        jQuery(element).closest('.tabs-content.flight-schedule.js-journey').after(html);
    }

  },

  validateEmail = function validateEmail(sEmail) {
    var filter = /^([\w-\.]+)@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$/;

    if (filter.test(sEmail)) {
      return true;
    }
    else {
      return false;
    }
  },

  bindBlurEvent = function bindBlurEvent(element){
    $(element).find(".PriceWatch-form-control").bind("blur", function(event, ui) { 
      var sEmail = $(element).find('#email').val();

      if (validateEmail(sEmail)) {
        $(element).find("#email").addClass('valid-email');
        $(element).find(".green-trick").css("display","block");
        $(element).find("#email").removeClass('invalid-email');
        $(element).find(".error-icon").css("display","none");
        $(element).find(".errors").css("display","none");
        $(element).find("#email").css("border","1px solid #00d400", "background-color","#FFF");
      } else {
        $(element).find("#email").removeClass('valid-email');
        $(element).find("#email").addClass('invalid-email');
        $(element).find(".green-trick").css("display","none");
        $(element).find(".error-icon").css("display","block");
        $(element).find(".errors").css("display","block");
        $(element).find("#email").css("border","1px solid red", "background-color","#FFF3F3");
      }
    });

    $(element).find(".PriceWatch-form-control").bind("focus", function(event, ui) {                          
      var sEmail = $(element).find('#email').val();
      $(element).find("#email").removeClass('invalid-email');
      $(element).find(".error-icon").css("display","none");
      $(element).find(".errors").css("display","none");  
      $(element).find("#email").css("border","1px solid #0692FC"); 
      $(element).find(".green-trick").css("display","none");        
    });
  },


  

  SelectFlightsPage = function SelectFlightsPage(element){
    console.log($(element).find('#btnValidate'));
    $(element).find('#btnValidate').click(function() { 
      var sEmail = $(element).find('#email').val();
      if ($.trim(sEmail).length == 0 || !validateEmail(sEmail)) {

        jQuery(element).find(".error-icon").css("display","block");
        jQuery(element).find(".errors").css("display","block");
        jQuery(element).find("#email").addClass('invalid-email');
        jQuery(element).find("#email").css("border","1px solid red", "background","#FFF3F3");
        jQuery(element).find(".PriceWatch-text-area").removeClass("Submit-text-area");
      }

       if (validateEmail(sEmail) && $.trim(sEmail).length > 0) {
      //if (site.scope.price_watch.validateEmailField('input#email')) {

        	site.scope.price_watch.submitPriceWatch();

	        jQuery(element).find(".submit-form").hide();
	        jQuery(element).find(".legal-copy").hide();
	        jQuery(element).find(".PriceWatch-text-area").text("Thanks for signing up! We'll start sending you great prices.").css("text-align", "center");
	        jQuery(element).find(".PriceWatchSubmit-img").css("display","block");
	        jQuery(element).find(".form-text").css("padding-top","5px");
	        jQuery(element).find(".PriceWatch-Padding").css("width","100%");

      } else {
        jQuery(element).find(".PriceWatch-text-area").removeClass("Submit-text-area");
      }
    });
  },

  FaresGuaranteedHide = function FaresGuaranteedHide(){
    jQuery($return).next().addClass('Fares-guaranteed-text');
    jQuery('.Fares-guaranteed-text').hide();
  },

  resizeScreen = function resizeScreen() {
    var screensize = document.documentElement.clientWidth;
    if (screensize  < 1150){
      jQuery('.legal-header .toggle-legal-show').before('<br/>');
    }
    $('.flight-schedule__selected-date ').addClass('small-8').css( 'margin',' 10px 0');
    $('.flight-schedule__selected-currency ').addClass('small-4').css( 'margin',' 10px 0');
  },

  	actionTracking = function actionTracking() {
	  	if(typeof(window.Jumpseat) == 'object' || window.Jumpseat !== undefined){
	      	window.Jumpseat.run('addListener', {  
	        	eventName: 'subscriptionCreated',  
	        	listener: onSubscriptionCreated  
	      	})

	      	function onSubscriptionCreated() {  
	        	actions.send('Deals_PriceWatch', '1');
	        	console.log('Subscription Success');
	      	}
	  	}
	};

  return {
    init: init
  };

}(window, document));

      window.jetstarOptimisation.PWCTAColor_SelectFlightsPage_SG.init();
    </script>
